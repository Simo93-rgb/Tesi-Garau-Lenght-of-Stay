\section{Contesto e Motivazione}

% ── SLIDE 3: Il Problema — LOS ───────────────────────────────
\begin{frame}{Il Problema: Length of Stay (LOS)}
  \textbf{Cos'è il LOS?}\\
  La \alert{durata della degenza ospedaliera} è un indicatore
  critico di efficienza clinica e sostenibilità economica.

  \vspace{1em}
  \begin{block}{Soglia Critica}
    Degenze $\ge 20$ giorni classificate come \textbf{long-LOS}:\\
    sovraccarico di letti, rischio nosocomiale, costi elevati.
  \end{block}
\end{frame}

% ── SLIDE 4: Limiti Process Mining ───────────────────────────
\begin{frame}{I Limiti dell'Approccio Classico}
  \textbf{Process Mining tradizionale:}
  \begin{itemize}
    \item Estrae grafi deterministici da log XES
    \item Modella flussi di cura e deviazioni standard
    \item Efficace per processi stabili e a bassa variabilità
  \end{itemize}

  \vspace{1em}
  \begin{alertblock}{Le Criticità sui Log Clinici}
    \begin{itemize}
      \item \textbf{Alta variabilità e rumore} nei log reali (distribuzione a coda lunga)
      \item \textbf{Dipendenze temporali} a lungo termine
      \item \textbf{Complessità semantica} delle azioni cliniche
    \end{itemize}
  \end{alertblock}

  \vspace{1em}
  \textbf{Conclusione:} Serve un modello capace di comprendere 
  il \alert{contesto bidirezionale} di una sequenza.
\end{frame}

% ── SLIDE 9: BERT + Focal Loss ────────────────────────────────
\begin{frame}{Il Modello: BERT e $\alpha$-Balanced Focal Loss }
  \begin{columns}[T]
    \begin{column}{0.50\textwidth}
      \textbf{Architettura:} \texttt{bert-base-uncased} + classification head
      \begin{itemize}
        \item 110M parametri, tokenizer WordPiece
        \item Perché non ClinicalBERT? \alert{Vocabulary mismatch}:\\
              nomenclature locali non standardizzate
        \item Setup (Stratified 5-Fold CV): \texttt{lr}=$5\text{e-}5$, batch=22, early stopping epoca 10
      \end{itemize}

      \vspace{0.4em}
      \begin{block}{Focal Loss ($\gamma=2.0$, $\alpha=[0.25, 0.75]$)}
        \[
          FL(p_t) = -\alpha_t\,(1-p_t)^{\gamma}\,\log(p_t)
        \]
        Il fattore $(1-p_t)^\gamma$ \textbf{abbatte il gradiente}
        degli esempi facili (classe 0 dominante), forzando BERT
        ad imparare i casi rari di long-LOS.
      \end{block}
    \end{column}

    \begin{column}{0.47\textwidth}
      \centering
      \includegraphics[width=\linewidth]{focal loss fig 1.png}
      {\small\itshape Al crescere di $\gamma$, la loss sugli esempi\\
      facili ($p_t \to 0$) crolla: il modello si \alert{concentra}\\
      sui casi complessi e rari. \\ \alert{\textbf{Attenzione}}: Un $\gamma$ eccessivo porta 
      il modello a ignorare i casi normali, rischiando di overfittare esclusivamente 
      sulle anomalie}
    \end{column}
  \end{columns}
\end{frame}


% ── SLIDE 7: La Ricerca dell'Embedding Ideale ─────────────────
\begin{frame}{La Ricerca dell'Embedding Ideale}
  Per interfacciare i log clinici testuali e temporali con il Transformer, 
  sono state testate \textbf{tre distinte filosofie di embedding}:

  \vspace{0.3em}
  \begin{columns}[T]
    \begin{column}{0.30\textwidth}
      \begin{block}{1. Array Concatenati}
        \small
        Doppio input parallelo:
        \begin{itemize}
            \item Array testuale (le azioni cliniche).
            \item Array numerico \textit{float} (i delta temporali in secondi).
        \end{itemize}
      \end{block}
    \end{column}

    \begin{column}{0.35\textwidth}
      \begin{block}{2. Binning Temporale}
        \small
        Discretizzazione \textit{data-driven} del tempo. Inserimento di token 
        speciali derivati dalla distribuzione empirica, ad esempio:
        \begin{itemize}
          \item \small \texttt{TIME\_DELTA\_SHORT\_} (0-1gg)
          \item \small \texttt{TIME\_DELTA\_MEDIUM\_} (1-7gg)
        \end{itemize}
      \end{block}
    \end{column}

    \begin{column}{0.30\textwidth}
      \begin{exampleblock}{3. Storytelling}
        \small
        Traduzione semantica in frasi inglesi strutturate. 
        Gli eventi e i vincoli cronologici generano paragrafi narrativi.
        \textbf{L'approccio rivelatosi vincente.}
      \end{exampleblock}
    \end{column}
  \end{columns}
\end{frame}

% ── SLIDE 8: L'Approccio Vincente: Il Paradigma ``Storytelling'' ──────────────
\begin{frame}{L'Approccio Vincente: Il Paradigma ``Storytelling''}
  \textbf{L'Intuizione:} tradurre la traccia in un \textbf{testo in linguaggio naturale} massimizza 
  la capacità di BERT di estrarre contesto bidirezionale.

  \vspace{0.0em}
  \begin{columns}[T]
    \begin{column}{0.48\textwidth}
      \begin{block}{Da Log XES\ldots}
        
        \texttt{case\_id: 5870415}\\
        \texttt{--> Complete abdomen ultrasound}\\
        \texttt{--> CT skull} \quad$\Delta = 0$~s\\
        \texttt{--> Complete abdomen ultrasound} \quad$\Delta = 728.160$~s
      \end{block}
    \end{column}
    
    \begin{column}{0.48\textwidth}
      \begin{exampleblock}{\ldots a Narrazione}
        \small\itshape
        ``The patient, 87 years old, underwent a series of
        examinations. \\
        After 0 seconds, \textbf{the following examinations
        were performed simultaneously}: Complete abdomen ultrasound
        and CT skull. \\
        After \textbf{728,160 seconds}, Complete abdomen
        ultrasound was performed.''
      \end{exampleblock}
    \end{column}
  \end{columns}

  \vspace{0.0em}
  \textbf{Requisiti Ingegneristici Algoritmo:}
  \begin{itemize}
      \item \textbf{Gestione Tempo:} Iniezione esatta del \textit{delta in secondi} (nessuna approssimazione).
      \item \textbf{Eventi Simultanei:} Aggregazione per evitare ridondanza e non sovraccaricare l'Attention di BERT con token identici.
  \end{itemize}
\end{frame}

% ── SLIDE 11: XAI — Black Box ──────────────────────────────────
\section{Explainable AI}

\begin{frame}{Il Problema della Black-Box in Sanità}
  \begin{columns}[T]
    \begin{column}{0.55\textwidth}
      \begin{alertblock}{Il Paradosso della Black-Box}
        Una rete neurale è un oracolo opaco: restituisce
        una probabilità, ma \alert{non spiega perché}.\\
        In sanità questa opacità è inaccettabile.
      \end{alertblock}

      {\footnotesize
      \textbf{Perché le tecniche classiche falliscono sul testo?}
      \begin{itemize}
        \item \textbf{Grad-CAM:} ideato per immagini (CNN), incompatibile con la \textit{Self-Attention}.
        \item \textbf{SmoothGrad / Blur IG:} sfocare un pixel continuo ha senso, ma perturbare casualmente un \textit{token discreto} genera vettori privi di significato semantico.
      \end{itemize}

      \textbf{Soluzione:} \alert{Integrated Gradients (IG)}\\
      Sostituisce la perturbazione con un'integrazione da una \textit{baseline} fissa, garantendo validità assiomatica sulle sequenze testuali. \par}
    \end{column}

    \begin{column}{0.42\textwidth}
      \centering
      \includegraphics[width=\linewidth]{comparison_with_a_dog.png}
      {\small\itshape I metodi classici (es. Vanilla Gradient) soffrono di\\ \textit{gradient shattering}. IG fornisce attribuzioni\\ stabili mediando queste oscillazioni.}
    \end{column}
  \end{columns}
\end{frame}

% ── SLIDE 12: Integrated Gradients ────────────────────────────
\begin{frame}{Integrated Gradients e la Baseline \texttt{[PAD]}}
  \begin{columns}[T]
    \begin{column}{0.55\textwidth}
      \textbf{Intuizione:} misurare il contributo di ogni token
      integrando i gradienti lungo un percorso logico dalla
      \alert{baseline} (assenza d'informazione) all'input reale.

      \vspace{0.4em}
      \begin{block}{Approssimazione dell'Integrale}
        \footnotesize
        L'integrale continuo non è calcolabile su reti profonde.\\
        Si utilizza un'approssimazione discreta a \textbf{$m$ step} 
        (frazioni spaziali equidistanti tra baseline e input).
      \end{block}

      \begin{block}{La Baseline: Lo "Zero Semantico"}
        \footnotesize
        $x' = $ vettore di token \texttt{[PAD]} (ID=0).\\
        Garantisce a BERT un punto d'origine privo di informazione 
        clinica, rispettando l'\textbf{Assioma di Sensibilità}.
      \end{block}
    \end{column}

    \begin{column}{0.42\textwidth}
      \begin{alertblock}{Verifica di Convergenza (Completeness)}
        \footnotesize
        \centering $\sum \text{Attribuzioni} \approx f(x) - f(x')$ \\[0.4em]
        \raggedright Garantisce che la somma spieghi \textit{esattamente} la differenza di predizione. \textbf{Tolleranza:} Errore Relativo $\le 5\%$.
      \end{alertblock}

      \vspace{0.0em}
      \begin{exampleblock}{Ottimizzazione Hardware (GPU)}
        \footnotesize
        Il default di libreria (Gauss-Legendre) satura la CPU.\\
        \textbf{Soluzione adottata: Riemann Trapezoid}
        \begin{itemize}
          \item Vettorializzazione 100\% su GPU.
          \item Tempi: da \alert{15~s} a \textbf{2.5~s / traccia}.
          \item Algoritmo adattivo per garantire la convergenza a tutte le tracce con $m \text{ steps} \in [1500,5500] \subset \mathbb{N}$
        \end{itemize}
      \end{exampleblock}
    \end{column}
  \end{columns}
\end{frame}

% ── SLIDE 13: Aggregazione Sub-Token ─────────────────────────
\begin{frame}{La Sfida: Aggregazione dei Sub-Token}
  \textbf{WordPiece frammenta} le parole cliniche in sub-token privi
  di significato autonomo. \\ \alert{\textbf{Serve ricostruirle}}.

  \vspace{0.5em}
  \begin{columns}[T]
    \begin{column}{0.52\textwidth}
      \textbf{Step A — da 768 dimensioni a scalar:}\\
      Il vettore IG viene collassato in un
      singolo valore per token.

      \vspace{0.4em}
      \textbf{Step B — root sum dei \texttt{\#\#} fragments:}\\
      I sub-token marcati \texttt{\#\#} sommano il loro score
      al token radice $\to$ parola clinica intera.

      \vspace{0.4em}
      Token artificiali \texttt{[CLS]}, \texttt{[SEP]} esclusi.

      \vspace{0.4em}
      \begin{exampleblock}{Risultato}
        Ogni \textbf{azione clinica reale} ottiene un
        unico score aggregato, leggibile dal clinico.
      \end{exampleblock}
    \end{column}
